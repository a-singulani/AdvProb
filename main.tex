% -------------------------------------------------
% Shared theorem counter (like amsthm), numbered within chapter
% This makes \label inside work normally.
% -------------------------------------------------
\newcounter{thm}[chapter]
\renewcommand{\thethm}{\thechapter.\arabic{thm}}

\NewDocumentEnvironment{theorem}{o}
{%
  \refstepcounter{thm}%
  \begin{tcolorbox}[mybox, mycolors=ThmNavy,
    title={Theorem~\thethm\IfValueT{#1}{ (\unexpanded{#1})}}%
  ]%
}
{\end{tcolorbox}}
